// /js/views/Thread.js
export default function ThreadView(params) {
  return {
    render: async () => `
      <h1>Thread ${params.id}</h1>
      <button id="loadMore">Load more</button>
    `,
    mount: () => {
      // view-specific code here
      const btn = document.querySelector("#loadMore");
      btn.addEventListener("click", onLoadMore);
    },
    unmount: () => {
      // cleanup (remove listeners, stop intervals, etc.)
      const btn = document.querySelector("#loadMore");
      btn?.removeEventListener("click", onLoadMore);
    }
  };
}

function onLoadMore() {
  console.log("loading more...");
}
